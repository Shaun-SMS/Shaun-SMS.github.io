# remove duplicates from initial stress series
def remove_duplicates(x):
    i = 0
    while i < len(x)-1:
        if x[i+1] == x[i]:
            del x[i+1]
        else:
            i += 1
    return x

# find turning points for each stress series
def turning_points(x):
    i = 1
    while i < len(x)-1:
        if((x[i-1]<x[i] and x[i+1]<x[i]) or (x[i-1]>x[i] and x[i+1]>x[i])):
            i += 1
        else:
            del x[i]
    return x

# rearrange the stress series to begin and end with the largest peak
def rearrange_tp(x):
    max_value = max(x)
    max_index = x.index(max_value)
    fst_half = x[:max_index+1]
    sec_half = x[max_index:]
    temp_list = sec_half + fst_half
    
    rearranged_list = remove_duplicates(temp_list)
    rearranged_list = turning_points(rearranged_list)
    return rearranged_list

# perform rainflow counting for clyces and stress ranges
def rainflow_counting(x):
    i = 1
    tot_cycle = 0
    stress_range = []
    while i < len(x)-1:
        delta_1 = abs(x[i] - x[i-1])
        delta_2 = abs(x[i] - x[i+1])
        if(delta_2 >= delta_1):
            tot_cycle += 1
            stress_range.append(round(delta_1,5))
            del x[i]
            del x[i-1]
            i = 1
        else:
            i += 1
    stress_range.sort()
    stress_range.insert(0, 'stress range (MPa)')
    return stress_range

# process the data to obtain the clyces and stress ranges
header = list(data.columns.values)
ecas = []        # equivalent constant amplitude stress range
for i in range(1, len(header)):
    a = data[header[i]].values.tolist()
    b = remove_duplicates(a)
    c = turning_points(b)
    d = rearrange_tp(c)
    e = rainflow_counting(d)

    # calculate equivalent constant amplitude stress range
    p = []
    for i in range(1, len(e)):
        p.append(e[i]**5)
        q = (sum(p)/(len(e)-1))**0.2
    ecas.append(round(q,5))

# calculate everage equivalent constant amplitude stress range of SBSG stress series
for i in range(1, len(ecas)):
    avg = (sum(ecas) - ecas[0])/(len(ecas)-1)
    avg = round(avg,5)

# compare equivalent constant amplitude stress range between LBSG and SBSG of the same location
diff = abs((ecas[0] - avg))/ecas[0]
percent = "{:.2%}".format(diff)
print(percent)
